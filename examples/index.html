function knightsTour() {
  // This demo plays on a 3x7 board attached to <div id="knights-tour">
  const knightsTour = create('board', { columns: 3, rows: 7 });
  knightsTour.addView('board', '#knights-tour');
  // On smaller screens the view of the board is transposed.
  knightsTour.addView('board', '#knights-tour-mobile', { transpose: true})

  const moves
    = 'c2 b4 c6 a7 b5 c7 a6 c5 b7 a5 c1 a2 c3 b1 a3 c4 b2 a4 b6'.split(' ');
  // Place the knight in the initial position then play the rest of the moves.
  return knightsTour.set('a1', 'N').then(() => {
    // ...then we can play through the rest of the moves.
    knightsTour.move('a1', moves);
  });
}
